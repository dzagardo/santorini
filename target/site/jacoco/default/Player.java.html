/**
 * Player class has 2 workers and a player number.
 * The Player constructor takes a number
 * for player number (1, 2, 3, 4), and passes it through
 * to the worker constructor to assign two workers
 * with an assigned player number.
 */
public class Player {

    private int playerNumber;
    private boolean hasMoved;
    private Worker workerOne;
    private Worker workerTwo;

    /**
     * Constructor for Player. Making trivial change.
     * @param num
     */
    public Player(int num) {
        playerNumber = num;
        this.workerOne = new Worker(num);
        this.workerTwo = new Worker(num);
    }

    public Worker getWorkerOne() {
        return workerOne;
    }

    public Worker getWorkerTwo() {
        return workerTwo;
    }

    public int getPlayerNumber() {
        return playerNumber;
    }

    public boolean getHasMoved() {
        return hasMoved;
    }

    public void hasMoved() {
        hasMoved = true;
    }

    public void endOfTurn() {
        hasMoved = false;
    }

    public void workerPosSetter(Worker w, int x, int y) {
        w.setPosition(x, y);
    }

    public int[][] workerGetPosHelper(Worker w) {
        return w.getPosition();
    }
}
