public class Board {

    private Space[][] board;
    private static final int BOARD_DIMENSION = 5;

    /**
     * New Board method. Creates a 5x5 board, using
     * a 2D array data structure.
     */
    public Board() {
        board = new Space[BOARD_DIMENSION][BOARD_DIMENSION];
        for (int i = 0; i < BOARD_DIMENSION; i++) {
            for (int j = 0; j < BOARD_DIMENSION; j++) {
                board[i][j] = new Space(i, j);
            }
        }
    }

    public Space build(int x, int y) {
        // validate the inputs
        // validate the move
        Space space = getSpace(x, y);
        space.buildTower();
        return space;
    }

    public Space[][] getBoard() {
        return board;
    }

    public Space getSpace(int x, int y) {
        return board[x][y];
    }

    public Space getSpace(int[][] position) {
        return board[position[0][0]][position[0][1]];
    }

    public void printBoardHeights() {
        for (int i = 0; i < BOARD_DIMENSION; i++) {
            for (int j = 0; j < BOARD_DIMENSION; j++) {
                System.out.print(board[i][j].getTowerHeight());
            }
            System.out.println();
        }
    }

    public void printBoardOccupancy() {
        for (int i = 0; i < BOARD_DIMENSION; i++) {
            for (int j = 0; j < BOARD_DIMENSION; j++) {
                System.out.print(board[i][j].getOccupancy());
            }
            System.out.println();
        }
    }
}
