public class Worker implements Piece {

    private int owner;
    private int workerNumber;
    private boolean hasMoved;
    private int[][] position = new int[1][2];

    // game has been started
    // worker has been set on the board
    //

    public Worker (int player) {
        this.owner = player;
    }

    @Override
    public void assignOwnership(int playerNumber, int workerNumber) {
        if (playerNumber <= 0 || workerNumber <= 0) {
            return;
        }
        this.owner = playerNumber;
        this.workerNumber = workerNumber;
    }

    public int getOwner() {
        return owner;
    }

    public void setPosition(int x, int y) {
        this.position[0][0] = x;
        this.position[0][1] = y;
    }

    public int[][] getPosition() {
        return position;
    }

    public boolean getHasMoved() {
        return hasMoved;
    }

    public void hasMoved() {
        hasMoved = true;
    }

    public void endOfTurn() {
        hasMoved = false;
    }

    public int getWorkerNumber() {
        return workerNumber;
    }

}
